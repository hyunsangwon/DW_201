### 페이징 알고리즘 step by step

```java
public class PageHandler {

   public static void main(String[] args) {

      int totalCount = 70; //전체 게시물 수
      int rowNum = 10; //한페이지에 보여주는 게시물 수
      int pageNum = 5; //현재 내가 보고 있는 페이지
      int blockNum = 3;//한 블록에 보여주는 페이지 수 (1p 2p 3p 4p 5p) == 1 block

      // step 1. 총 페이지 수 구하기
      // 전체 게시물 수와 한 페이지에 몇개를 보여줄지를 정하면 총 페이지 수가 나옴.
      int totalPage = totalCount / rowNum;
      if(totalCount % rowNum > 0) ++totalPage;
      System.out.println("총 페이지 수 => "+totalPage);

      //step 2. 현재 블록 알아내기
      int nowBlock = pageNum / blockNum;
      //System.out.println(block);
      if(pageNum % blockNum > 0) {
         ++nowBlock;
      }
      System.out.println("현재 블록은 => "+nowBlock);

      //step 3. 마지막 블록 알아내기
      int lastBlock = totalCount / (blockNum * rowNum);
      if(totalCount % (blockNum * rowNum) > 0) {
         ++lastBlock;
      }
      System.out.println("마지막 블록은 => "+lastBlock);

      //step 4. 이전,다음 유무 판단
      boolean isPrev = true; //이전 버튼
      boolean isNext = true; //다음 버튼

      if(lastBlock == 1){//블록이 하나이므로 이전, 다음 버튼 존재 X
        isPrev = false;
        isNext = false;
      }
      if(lastBlock > 1 && lastBlock == nowBlock){//마지막 블록이므로 이전 버튼 존재
        isPrev = true;
        isNext = false;
      }
      if(lastBlock > 1 && pageNum <= blockNum){
        isPrev = false;
        isNext = true;
      }
      System.out.println("이전 버튼 => "+isPrev);
      System.out.println("다음 버튼 => "+isNext);

      //step 5. 현재 페이지와 마지막 페이지 구하기
      int startPage = 0;
      int endPage = 0;
      startPage = (nowBlock * blockNum) - (blockNum - 1);
      endPage = startPage + (blockNum - 1);
      if(lastBlock == nowBlock) {
        endPage = totalPage;
      }
      System.out.println("현재 페이지 => "+startPage);
      System.out.println("마지막 페이지 => "+endPage);
    }
}
```
